  <script>
    const commonPorts = {
      22: "SSH",
      80: "HTTP",
      443: "HTTPS",
      3389: "RDP"
    };

    function scanDevice() {
      const ip = document.getElementById("ip").value;
      const ports = document.getElementById("port").value.split(",").map(p => p.trim());
      const tbody = document.querySelector("#results tbody");
      tbody.innerHTML = "";

      ports.forEach(port => {
        const service = commonPorts[port] || "Unknown";
        const status = Math.random() > 0.4 ? "Open" : "Closed";
        const row = `
          <tr>
            <td>${port}</td>
            <td style="color: ${status === 'Open' ? '#0f0' : '#f00'}">${status}</td>
            <td>${service}</td>
            <td><button ${status === 'Open' ? '' : 'disabled'} onclick="alert('Connecting to ${ip}:${port}')">Connect</button></td>
          </tr>
        `;
        tbody.innerHTML += row;
      });
    }
  </script>
