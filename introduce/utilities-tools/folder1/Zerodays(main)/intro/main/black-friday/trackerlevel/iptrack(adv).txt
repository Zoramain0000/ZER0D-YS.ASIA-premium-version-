  <script>
    let map = L.map('map').setView([12.8797, 121.7740], 5);
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '¬© OpenStreetMap contributors'
    }).addTo(map);
    let marker;
    let currentCoords = null;

    async function searchLocation() {
      const place = document.getElementById("placeInput").value.trim();
      const output = document.getElementById("infoOutput");
      const checkBtn = document.getElementById("checkBtn");

      if (!place) {
        alert("Please enter a location.");
        return;
      }

      const geocodeURL = `https://api.opencagedata.com/geocode/v1/json?q=${encodeURIComponent(place)}&key=c9600f6ca6ed4312a533fc58bbcfd371`;

      try {
        const geoRes = await fetch(geocodeURL);
        const geoData = await geoRes.json();

        if (!geoData.results.length) {
          output.textContent = "‚ùå Location not found.";
          return;
        }

        const { lat, lng } = geoData.results[0].geometry;
        currentCoords = { lat, lng }; // Store for later

        if (marker) map.removeLayer(marker);
        map.setView([lat, lng], 10);
        marker = L.marker([lat, lng]).addTo(map)
          .bindPopup(`<b>${place}</b><br>Coordinates: ${lat}, ${lng}`)
          .openPopup();

        output.textContent = `
üìç Location Found: ${place}
üõ∞ Coordinates    : ${lat}, ${lng}
‚ÑπÔ∏è Click "Check IP Info" for simulated IP
        `;
        checkBtn.style.display = "block";

      } catch (err) {
        output.textContent = "‚ùå Error fetching geolocation.";
        console.error(err);
      }
    }

    async function checkIPInfo() {
      const output = document.getElementById("infoOutput");
      const safe = val => val || "N/A";

      if (!currentCoords) {
        output.textContent = "‚ö†Ô∏è No coordinates set.";
        return;
      }

      try {
        const ipRes = await fetch(`https://ip-api.com/json/?fields=query,city,regionName,country,isp,org,as`);
        const ipData = await ipRes.json();

        output.textContent += `
üì° Simulated IP   : ${safe(ipData.query)}
üåê ISP / Org      : ${safe(ipData.isp)} / ${safe(ipData.org)}
üìç Region         : ${safe(ipData.city)}, ${safe(ipData.regionName)}, ${safe(ipData.country)}
üî¢ ASN            : ${safe(ipData.as)}
        `;
      } catch (err) {
        output.textContent += "\n‚ùå Error fetching IP data.";
        console.error(err);
      }
    }
